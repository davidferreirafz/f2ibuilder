Lendo Métricas - Metrics Read

//Global our member of class
char largura[256];

void carregarArquivo(std::string fileimage)
{
 FILE *file;
 std::string txt="";
 imagem = IMG_Load(fileimage.c_str());

 if (imagem!=NULL){
 txt=ARQUIVO.substr(0,fileimage.length()-4);
 txt+=".dat";
 file = fopen(txt.c_str(),"rb");
 fread(&largura, 256, 1, file);
 fclose(file);
 }
}

Escrevendo com Métricas - Writting with metrics

void escrever(const char * PALAVRA, int X, int Y)
{
 int i,t=strlen(PALAVRA);
 unsigned char l;
 posicao.x=X; //screen x
 posicao.y=Y; //screen y

 for (i=0; i<t; i++){
 l=PALAVRA[i];
 tamanho.x=(l%16)*dimensaoQuadro.w; tamanho.w=largura[l];
 tamanho.y=(l/16)*dimensaoQuadro.h; tamanho.h=dimensaoQuadro.h;
 gsScreen->blitSurface(imagem,&tamanho,&posicao);
 posicao.y=Y;
 posicao.x=posicao.x+largura[l];
 }
}
